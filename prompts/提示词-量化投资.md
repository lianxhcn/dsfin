```python
import os
import baostock as bs
import pandas as pd
from datetime import datetime

# =========================
# 1. 创建目录：保存到 data/stock/
# =========================
os.makedirs("data/stock", exist_ok=True)

# =========================
# 2. 登录 baostock
# =========================
lg = bs.login()
print("login respond error_code:", lg.error_code)
print("login respond error_msg:", lg.error_msg)

# =========================
# 3. 时间范围
# =========================
start_date = "2020-01-01"
end_date = datetime.today().strftime("%Y-%m-%d")

# =========================
# 4. 自选10只股票
# 覆盖至少5个行业，每个行业不超过2只
# =========================
stocks = {
    "600036": {"bs_code": "sh.600036", "name": "招商银行", "industry": "银行"},
    "601633": {"bs_code": "sh.601633", "name": "长城汽车", "industry": "汽车"},
    "000063": {"bs_code": "sz.000063", "name": "中兴通讯", "industry": "通讯"}
}
# =========================
# 5. 下载股票后复权日线数据
# 保存到 data/stock/stock_股票代码.csv
# =========================
for code, info in stocks.items():
    bs_code = info["bs_code"]
    print(f"\n正在下载 {code} {info['name']} ({info['industry']})")

    rs = bs.query_history_k_data_plus(
        bs_code,
        "date,code,open,high,low,close,volume,amount",
        start_date=start_date,
        end_date=end_date,
        frequency="d",
        adjustflag="3"   # 后复权；若你本地验证后发现定义不同，可改为 "2"
    )

    data_list = []
    while (rs.error_code == '0') and rs.next():
        data_list.append(rs.get_row_data())

    df = pd.DataFrame(data_list, columns=rs.fields)

    # 转换数值列
    numeric_cols = ["open", "high", "low", "close", "volume", "amount"]
    for col in numeric_cols:
        df[col] = pd.to_numeric(df[col], errors="coerce")

    # 重命名列，和作业要求更一致
    df = df.rename(columns={
        "date": "日期",
        "code": "股票代码",
        "open": "开盘价",
        "close": "收盘价",
        "high": "最高价",
        "low": "最低价",
        "volume": "成交量",
        "amount": "成交额"
    })

    # 保存到 data/stock/
    save_path = f"data/stock/stock_{code}.csv"
    df.to_csv(save_path, index=False, encoding="utf-8-sig")
    print(f"已保存: {save_path}, 共 {len(df)} 行")
```
